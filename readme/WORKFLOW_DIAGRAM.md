```typescript
// Can mark entry?
!hasEnteredToday() && !isSubmittedToday()

// Can mark exit?
hasEnteredToday() && !hasExitedToday() && !isSubmittedToday()

// Show todos?

log.date === today && log.isSubmitted === true

// Has entered today?

// Has exited today?
```
